# 3. Pattern Recognition

## What is Pattern Recognition?

Identifying repeating sequences or structures in data.

---

## Visual Patterns

### Pattern Type 1: Repeating Sequence

```
Pattern: A B C A B C A B C

Index:   0 1 2 3 4 5 6 7 8
Value:   A B C A B C A B C
         └─┬─┘ └─┬─┘ └─┬─┘
         cycle cycle cycle

Period = 3
```

```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    vector<int> pattern = {1, 2, 3};
    int n = 10;
    
    cout << "Extended pattern: ";
    for (int i = 0; i < n; i++) {
        cout << pattern[i % 3] << " ";
    }
    // Output: 1 2 3 1 2 3 1 2 3 1
    
    return 0;
}
```

### Pattern Type 2: Alternating Values

```
Pattern: A B A B A B

Index:   0 1 2 3 4 5
Value:   A B A B A B
         ↓ ↓ ↓ ↓ ↓ ↓
        ODD positions: A
        EVEN positions: B
```

```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n = 8;
    int A = 5, B = 7;
    
    vector<int> arr(n);
    
    for (int i = 0; i < n; i++) {
        arr[i] = (i % 2 == 0) ? A : B;
    }
    
    cout << "Alternating pattern: ";
    for (int x : arr) cout << x << " ";
    // Output: 5 7 5 7 5 7 5 7
    
    return 0;
}
```

### Pattern Type 3: All Same

```
Pattern: X X X X X X

Index:   0 1 2 3 4 5
Value:   X X X X X X
         └───────────┘
         All identical
```

```cpp
#include <iostream>
#include <vector>
using namespace std;

bool allSame(vector<int>& arr) {
    for (int i = 1; i < arr.size(); i++) {
        if (arr[i] != arr[0]) {
            return false;
        }
    }
    return true;
}

int main() {
    vector<int> arr1 = {5, 5, 5, 5};
    vector<int> arr2 = {5, 5, 3, 5};
    
    cout << "arr1 all same: " << (allSame(arr1) ? "Yes" : "No") << "\n";
    cout << "arr2 all same: " << (allSame(arr2) ? "Yes" : "No") << "\n";
    
    return 0;
}
```

### Pattern Type 4: Palindrome

```
Pattern: A B C B A

Index:   0 1 2 3 4
Value:   A B C B A
         ↓       ↓
         ↓   ↓   ↓
         └───┴───┘
         Mirror symmetry
```

```cpp
#include <iostream>
#include <vector>
using namespace std;

bool isPalindrome(vector<int>& arr) {
    int n = arr.size();
    for (int i = 0; i < n / 2; i++) {
        if (arr[i] != arr[n - 1 - i]) {
            return false;
        }
    }
    return true;
}

int main() {
    vector<int> arr1 = {1, 2, 3, 2, 1};
    vector<int> arr2 = {1, 2, 3, 4, 5};
    
    cout << "arr1 is palindrome: " << (isPalindrome(arr1) ? "Yes" : "No") << "\n";
    cout << "arr2 is palindrome: " << (isPalindrome(arr2) ? "Yes" : "No") << "\n";
    
    return 0;
}
```

## Detecting Pattern Period

```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    vector<int> arr = {1, 2, 3, 1, 2, 3, 1, 2, 3};
    
    // Check for period
    for (int period = 1; period <= arr.size() / 2; period++) {
        bool is_periodic = true;
        
        for (int i = period; i < arr.size(); i++) {
            if (arr[i] != arr[i % period]) {
                is_periodic = false;
                break;
            }
        }
        
        if (is_periodic) {
            cout << "Pattern repeats with period: " << period << "\n";
            break;
        }
    }
    
    return 0;
}
```

## Checking Specific Patterns

### Arithmetic Sequence

```
Pattern: 2 5 8 11 14
         +3 +3 +3 +3
```

```cpp
#include <iostream>
#include <vector>
using namespace std;

bool isArithmetic(vector<int>& arr) {
    if (arr.size() < 2) return true;
    
    int diff = arr[1] - arr[0];
    
    for (int i = 2; i < arr.size(); i++) {
        if (arr[i] - arr[i-1] != diff) {
            return false;
        }
    }
    return true;
}

int main() {
    vector<int> arr = {2, 5, 8, 11, 14};
    
    if (isArithmetic(arr)) {
        cout << "Arithmetic sequence\n";
    }
    
    return 0;
}
```

### Geometric Sequence

```
Pattern: 2 6 18 54
         ×3 ×3 ×3
```

```cpp
#include <iostream>
#include <vector>
using namespace std;

bool isGeometric(vector<int>& arr) {
    if (arr.size() < 2) return true;
    if (arr[0] == 0) return false;
    
    int ratio = arr[1] / arr[0];
    
    for (int i = 2; i < arr.size(); i++) {
        if (arr[i-1] == 0 || arr[i] / arr[i-1] != ratio) {
            return false;
        }
    }
    return true;
}

int main() {
    vector<int> arr = {2, 6, 18, 54};
    
    if (isGeometric(arr)) {
        cout << "Geometric sequence\n";
    }
    
    return 0;
}
```

## Finding Repeating Subsequences

```cpp
#include <iostream>
#include <vector>
#include <string>
using namespace std;

int main() {
    vector<int> arr = {1, 2, 1, 2, 1, 2};
    
    // Check if array is made of repeated sub-pattern
    for (int len = 1; len <= arr.size() / 2; len++) {
        if (arr.size() % len != 0) continue;
        
        bool matches = true;
        for (int i = len; i < arr.size(); i++) {
            if (arr[i] != arr[i % len]) {
                matches = false;
                break;
            }
        }
        
        if (matches) {
            cout << "Repeating pattern of length " << len << ": ";
            for (int i = 0; i < len; i++) {
                cout << arr[i] << " ";
            }
            cout << "\n";
            break;
        }
    }
    
    return 0;
}
```
